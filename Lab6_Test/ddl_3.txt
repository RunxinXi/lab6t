DROP TABLE IF EXISTS LAWYERS;
CREATE TABLE IF NOT EXISTS LAWYERS (
	ID INTEGER,
	NAME TEXT,
	HOURLYRATE FLOAT
);

DROP TABLE IF EXISTS TIMESHEETS;
CREATE TABLE IF NOT EXISTS TIMESHEETS (
	LAWYERID INTEGER,
	CLIENTID INTEGER,
	YEAR INTEGER,
	MONTH INTEGER,
	DAY INTEGER,
	DESCRIPTION TEXT,
	HOURSBILLED FLOAT,
	PRIORITY BOOLEAN
);

DROP TABLE IF EXISTS INVOICES;
CREATE TABLE IF NOT EXISTS INVOICES (
	CLIENTID INTEGER,
	LAWYERID INTEGER,
	YEAR INTEGER,
	MONTH INTEGER,
	TOTAL FLOAT
);
